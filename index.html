Q1 = Find the number of digits

        const Find_Digits = (N) => 
{
  
                const numStr = N.toString();
            
                
                let count = 0;
            
                
                for (const ch of numStr) {
                    
                    if (/[0-9]/.test(ch)) {
                        count++;
                    }
                }
            
                
                return count;
};

{

Q7 = Check whether a Number is a prime or not

    
const Prime_Check = (N) => 
{
	 if (N < 2) {
                return "NO";
              }
              for (let i = 2; i <= Math.sqrt(N); i++) {
                if (N % i === 0) {
                  return "NO";
                }
              }
              return "YES";
	
};
